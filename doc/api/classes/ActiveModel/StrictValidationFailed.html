Class ActiveModel::StrictValidationFailed < StandardError

  • /usr/local/rvm/gems/ruby-2.0.0-p0/gems/activemodel-4.0.0/lib/active_model/errors.rb

Raised when a validation cannot be corrected by end users and are considered exceptional.

class Person
  include ActiveModel::Validations

  attr_accessor :name

  validates_presence_of :name, strict: true
end

person = Person.new
person.name = nil
person.valid?
# => ActiveModel::StrictValidationFailed: Name can't be blank
