def method_missing(symbol, &block)
  mime_constant = Mime.const_get(symbol.upcase)

  if Mime::SET.include?(mime_constant)
    AbstractController::Collector.generate_method_for_mime(mime_constant)
    send(symbol, &block)
  else
    super
  end
end
