def trigger(files)
  trigger_files = Hash.new { |h,k| h[k] = Hash.new { |h2,k2| h2[k2] = [] } }

  files.each do |file, state|
    @regex_matchers.each do |pattern, block|
      trigger_files[block][state] << file if pattern === file
    end
  end

  trigger_files.each do |block, payload|
    block.call payload
  end
end
