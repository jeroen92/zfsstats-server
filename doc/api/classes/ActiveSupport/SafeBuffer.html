def %(args)
  args = Array(args).map do |arg|
    if !html_safe? || arg.html_safe?
      arg
    else
      ERB::Util.h(arg)
    end
  end

  self.class.new(super(args))
end

def [](*args)
  if args.size < 2
    super
  else
    if html_safe?
      new_safe_buffer = super
      new_safe_buffer.instance_eval { @html_safe = true }
      new_safe_buffer
    else
      to_str[*args]
    end
  end
end
